#### **优化 4: 技能缓存策略**

```javascript
// 优化 skills-hook.js 加载逻辑
class QwenSkillsHook {
  constructor() {
    this.skillCache = new Map();
    this.cacheExpiry = new Map();
    this.TTL = 600000; // 10 分钟
  }

  async loadSkill(skillName) {
    const now = Date.now();
    const cached = this.skillCache.get(skillName);
    
    if (cached && now < this.cacheExpiry.get(skillName)) {
      return cached; // 缓存命中
    }
    
    const skill = await this._loadFromDisk(skillName);
    this.skillCache.set(skillName, skill);
    this.cacheExpiry.set(skillName, now + this.TTL);
    return skill;
  }
}
```
