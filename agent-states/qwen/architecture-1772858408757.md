```javascript
// 建议：事件驱动 + 优先级队列
class AgentLoop {
  constructor() {
    this.taskQueue = new PriorityQueue();
    this.circuitBreaker = new CircuitBreaker();
  }

  async processTask(task) {
    while (!this.taskQueue.isEmpty() && !this.circuitBreaker.isOpen()) {
      const nextTask = this.taskQueue.dequeue();
      const result = await this.executeWithTimeout(nextTask);
      
      if (result.done) break;
      if (result.followUp) this.taskQueue.enqueue(result.followUp, result.priority);
    }
  }
}
```

**收益：**
- 降低循环复杂度 O(n) → O(log n)
- 支持动态优先级调度
- 自然支持任务取消

#### 2. **实现断路器模式**

```javascript
class CircuitBreaker {
  constructor(options = {}) {
    this.failureThreshold = options.failureThreshold || 5;
    this.resetTimeout = options.resetTimeout || 30000;
    this.failures = 0;
    this.state = 'CLOSED';
    this.lastFailureTime = null;
  }

  async execute(fn) {
    if (this.state === 'OPEN') {
      if (Date.now() - this.lastFailureTime > this.resetTimeout) {
        this.state = 'HALF_OPEN';
      } else {
        throw new Error('Circuit breaker is OPEN');
      }
    }

    try {
      const result = await fn();
      if (this.state === 'HALF_OPEN') this.state = 'CLOSED';
      this.failures = 0;
      return result;
    } catch (error) {
      this.failures++;
      this.lastFailureTime = Date.now();
      if (this.failures >= this.failureThreshold) {
        this.state = 'OPEN';
      }
      throw error;
    }
  }
}
```

```javascript
class MemoryStore {
  constructor(baseDir) {
    this.memoryDir = path.join(baseDir, 'memory');
    this.sessionsFile = path.join(this.memoryDir, 'sessions.jsonl');
    this.index = new Map();  // 内存索引
    this.cache = new LRUCache({ max: 100 });
  }

  async storeSession(session) {
    const line = JSON.stringify({ timestamp: Date.now(), session }) + '\n';
    await fs.promises.appendFile(this.sessionsFile, line);
    
    // 更新索引
    this.index.set(session.id, { 
      offset: this.currentOffset, 
      timestamp: Date.now(),
      task: session.task 
    });
  }

  async retrieveMemories(query, limit = 10) {
    // 1. 检查缓存
    const cached = this.cache.get(query);
    if (cached) return cached;

    // 2. 使用索引过滤
    const relevantIds = await this.searchIndex(query);
    
    // 3. 只读取相关记录
    const sessions = await this.readSessions(relevantIds);
    
    // 4. 更新缓存
    this.cache.set(query, sessions);
    return sessions;
  }
}
```
