class QwenSkillsHook {
  constructor() {
    this.skillsDir = path.join('C:/Users/WIN10/.stigmergy/skills');
    this.loadedSkills = new Map();
    this.skillIndex = null;  // 技能索引缓存
  }

  async initialize() {
    await this.buildSkillIndex();  // 只构建索引，不加载内容
  }

  async buildSkillIndex() {
    // 快速扫描，只读取必要信息
    const skills = [];
    const dirs = await fs.promises.readdir(this.skillsDir, { withFileTypes: true });
    
    for (const dirent of dirs) {
      if (dirent.isDirectory()) {
        const skillPath = path.join(this.skillsDir, dirent.name);
        const skillMdPath = path.join(skillPath, 'SKILL.md');
        
        if (await fs.promises.access(skillMdPath).then(() => true).catch(() => false)) {
          const frontmatter = await this.readFrontmatter(skillMdPath);
          skills.push({
            name: dirent.name,
            path: skillPath,
            description: frontmatter?.description,
            loaded: false
          });
        }
      }
    }
    
    this.skillIndex = skills;
  }

  async loadSkill(skillName) {
    // 懒加载：只在需要时加载
    if (this.loadedSkills.has(skillName)) {
      return this.loadedSkills.get(skillName);
    }

    const skillInfo = this.skillIndex.find(s => s.name === skillName);
    if (!skillInfo) return null;

    const skill = await this._loadSkillImplementation(skillInfo.path);
    this.loadedSkills.set(skillName, skill);
    skillInfo.loaded = true;
    return skill;
  }
}

class CircularBuffer {
  constructor(maxSize) {
    this.maxSize = maxSize;
    this.buffer = [];
  }

  push(item) {
    this.buffer.push(item);
    if (this.buffer.length > this.maxSize) {
      this.buffer.shift();
    }
  }

  slice(start) {
    return this.buffer.slice(start);
  }

class QwenSafeExecutor {
  constructor() {
    // 预编译正则表达式
    this.blockedCommands = [
      { pattern: /superpowers:.*[^a-zA-Z0-9_-]/i, reason: 'invalid_chars' },
      { pattern: /stigmergy.*superpowers:.*/i, reason: 'stigmergy_superpowers' },
      { pattern: /Bash\(["'])stigmergy.*superpowers:.*/i, reason: 'bash_superpowers' },
    ];
    
    // 缓存安全超级能力集合
    this.safeSuperpowersSet = new Set([
      'using-git-worktrees',
      'writing-plans',
      // ...
    ]);
  }

  isCommandSafe(command) {
    // 快速路径：检查是否是已知的安全模式
    if (this.cachedResults.has(command)) {
      return this.cachedResults.get(command);
    }

    // 使用 Set O(1) 查找
    const superpowerMatch = command.match(/superpowers:([a-zA-Z0-9_-]+)/i);
    if (superpowerMatch) {
      const isSafe = this.safeSuperpowersSet.has(superpowerMatch[1].toLowerCase());
      this.cachedResults.set(command, isSafe);
      return isSafe;
    }

    // 正则检查
    for (const { pattern, reason } of this.blockedCommands) {
      if (pattern.test(command)) {
        this.cachedResults.set(command, false);
        return false;
      }
    }

    this.cachedResults.set(command, true);
    return true;
  }
}

async function safeReadFile(filePath, maxSize = Infinity) {
  try {
    const content = await fs.promises.readFile(filePath, 'utf8');
