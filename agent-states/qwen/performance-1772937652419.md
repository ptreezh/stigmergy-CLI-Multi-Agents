#### 优化 1: 技能缓存与懒加载

```javascript
class QwenSkillsHook {
  constructor() {
    this.skillsDir = path.join(os.homedir(), '.stigmergy/skills');
    this.loadedSkills = new Map();
    this.skillCache = new Map();  // 新增：技能内容缓存
    this.cacheTimestamp = new Map();  // 新增：缓存时间戳
    this.CACHE_TTL = 5 * 60 * 1000;  // 5 分钟缓存
    this.initialized = false;  // 新增：初始化标志
  }

  async initialize() {
    if (this.initialized) return;  // 防止重复初始化
    console.log('[QWEN_SKILLS] Initializing skills hook...');
    await this.loadSkills();
    this.initialized = true;
  }

  async loadSkill(skillPath) {
    const cacheKey = `skill:${skillPath}`;
    
    // 检查缓存是否有效
    if (this.skillCache.has(cacheKey)) {
      const timestamp = this.cacheTimestamp.get(cacheKey);
      if (Date.now() - timestamp < this.CACHE_TTL) {
        return this.skillCache.get(cacheKey);
      }
    }

    // 原有加载逻辑...
    const skill = await this._loadSkillFromDisk(skillPath);
    
    // 更新缓存
    this.skillCache.set(cacheKey, skill);
    this.cacheTimestamp.set(cacheKey, Date.now());
    
    return skill;
  }

  // 新增：清理过期缓存
  cleanupCache() {
    const now = Date.now();
    for (const [key, timestamp] of this.cacheTimestamp.entries()) {
      if (now - timestamp > this.CACHE_TTL) {
        this.skillCache.delete(key);
        this.cacheTimestamp.delete(key);
      }
    }
  }
}
```

```javascript
// qwen_safe_executor.js
class QwenSafeExecutor {
  constructor() {
    // 预编译正则表达式
    this.blockedCommands = [
      /superpowers:.*[^a-zA-Z0-9_-]/i,
      /stigmergy.*superpowers:.*/i,
      /Bash\(["'])stigmergy.*superpowers:.*/i,
      /node.*stigmergy.*superpowers:.*/i,
      /require\(['"].*superpowers.*['"]\)/i,
    ].map(pattern => ({ pattern, compiled: new RegExp(pattern.source, pattern.flags) }));

    // 缓存替换正则
    this.superpowerCallRegex = /(Use|Try|Call|Execute)\s+superpowers:([a-zA-Z0-9_-]+)/gi;
    this.stigmergyBlockRegex = /Bash\(["'])stigmergy\s+.*superpowers:.*?(\1)/gi;
  }

  isCommandSafe(command) {
    for (const { compiled } of this.blockedCommands) {
      if (compiled.test(command)) {
        return false;
      }
    }
    return true;
  }

  filterUnsafeCommands(text) {
    if (!text || typeof text !== 'string') return text;

    // 使用缓存的正则表达式
    return text
      .replace(this.superpowerCallRegex, (match, action, skillName) => {
        if (this.safeSuperpowers.includes(skillName.toLowerCase())) {
          return match;
        }
        return `// SAFETY NOTICE: "${skillName}" not available`;
      })
      .replace(this.stigmergyBlockRegex, 
        '"// SAFETY BLOCKED: stigmergy superpowers call blocked"'
      );
  }
}
```

#### 优化 3: 异步文件读取

```javascript
// 替换同步读取为异步
async loadSkills() {
  try {
    if (!fs.existsSync(this.skillsDir)) return;

    const skillDirs = await fs.promises.readdir(this.skillsDir, { withFileTypes: true })
      .then(entries => entries.filter(d => d.isDirectory()).map(d => d.name));

    // 并行加载所有技能
    const loadPromises = skillDirs.map(async (skillName) => {
      const skillPath = path.join(this.skillsDir, skillName);
      const skillMdPath = path.join(skillPath, 'SKILL.md');
      
      if (fs.existsSync(skillMdPath)) {
        const skill = await this.loadSkill(skillPath);
        if (skill) {
          this.loadedSkills.set(skillName, skill);
        }
      }
    });

    await Promise.all(loadPromises);
    console.log(`[QWEN_SKILLS] Loaded ${this.loadedSkills.size} skills`);
  } catch (error) {
    console.error('[QWEN_SKILLS] Failed to load skills:', error);
  }
}
```
