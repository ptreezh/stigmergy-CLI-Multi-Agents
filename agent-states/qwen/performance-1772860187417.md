### ⚠️ 识别的性能瓶颈

#### 1. **文件 I/O 效率问题**
```javascript
// 当前实现 - 低效
const files = fs.readdirSync(historyDir).sort().reverse().slice(0, 10);
const recentSessions = files.map((f) => {
  const content = fs.readFileSync(path.join(historyDir, f), "utf-8");
  return { file: f, content: content.substring(0, 5000) };
});
```
**问题**：
- 同步读取所有文件，阻塞主线程
- 读取完整内容后才截断（应先截断再处理）
- 无缓存机制，每次重复读取

#### 2. **LLM Token 浪费**
```javascript
// 当前：每个会话截取 5000 字符，可能包含大量无关内容
content.substring(0, 5000)
```
**问题**：
- 未过滤元数据、命令输出等噪声
- 可能超出 LLM 上下文限制
- Token 成本高

#### 3. **搜索策略低效**
```javascript
// 使用 DuckDuckGo HTML 界面（不稳定且慢）
curl -s "https://html.duckduckgo.com/html/?q=${query}"
```
**问题**：
- HTML 解析不可靠
- 无搜索结果质量评估
- 可能触发反爬虫

#### 4. **索引文件竞争条件**
```javascript
// 读写索引文件无锁保护
const index = fs.existsSync(indexFile)
  ? JSON.parse(fs.readFileSync(indexFile, "utf-8"))
  : [];
index.unshift(...);
fs.writeFileSync(indexFile, JSON.stringify(index.slice(0, 50), null, 2));
```
**问题**：
- 并发写入时可能丢失数据
- 无原子操作保证

#### 5. **无增量处理**
- 每次都重新分析所有历史会话
- 未利用之前的反思结果
- 重复 LLM 调用

#### 优化 1：异步流式文件处理
```javascript
const { readFile } = require('fs/promises');
const { createReadStream } = require('fs');

// 使用流式读取 + 异步并发
async function readRecentSessions(historyDir, limit = 10) {
  const files = await fs.promises.readdir(historyDir);
  const sorted = files.sort().reverse().slice(0, limit);
  
  const sessions = await Promise.all(
    sorted.map(async (f) => {
      const stream = createReadStream(path.join(historyDir, f), {
        highWaterMark: 5000
      });
      let content = '';
      for await (const chunk of stream) {
        content += chunk;
        if (content.length >= 5000) break;
      }
      return { file: f, content: content.substring(0, 4500) }; // 留余量给标记
    })
  );
  return sessions;
}
```

```javascript
function filterSessionContent(content) {
  // 移除命令输出、错误堆栈等噪声
  const patterns = [
    /^(\$|>) .*/gm,           // 命令提示符
    /^Error:.*/gm,            // 错误信息
    /^\[.*\].*/gm,            // 时间戳
    /```[\s\S]*?```/g         // 代码块（可选保留）
  ];
  
  let filtered = content;
  patterns.forEach(p => filtered = filtered.replace(p, ''));
  
  // 提取关键对话轮次
  const turns = filtered.split(/\n{2,}/).filter(t => t.trim().length > 50);
  return turns.slice(-20).join('\n\n'); // 只保留最近 20 个有效回合
}
```

#### 优化 4：原子写入与文件锁
```javascript
const { writeFile, rename } = require('fs/promises');
const { randomUUID } = require('crypto');

async function atomicAppendIndex(reflectionDir, entry) {
  const indexFile = path.join(reflectionDir, 'index.json');
  const tempFile = path.join(reflectionDir, `index.${randomUUID()}.tmp`);
  
  // 使用文件锁（需要 proper-lockfile 包）
  const lock = await lockfile.lock(indexFile, { retries: 3 });
  try {
    const index = JSON.parse(await readFile(indexFile, 'utf-8') || '[]');
    index.unshift(entry);
    await writeFile(tempFile, JSON.stringify(index.slice(0, 50), null, 2));
    await rename(tempFile, indexFile); // 原子替换
  } finally {
    await lockfile.unlock(lock);
  }
}
```

#### 优化 5：增量反思与缓存
```javascript
class ReflectionCache {
  constructor(cacheDir) {
    this.cacheDir = cacheDir;
    this.cache = this.loadCache();
  }
  
  async getUnprocessedSessions(cliHistory) {
    const processed = new Set(this.cache.processedSessions);
    return cliHistory.filter(s => !processed.has(s.file));
  }
  
  async updateCache(newReflection) {
    this.cache.reflections.push(newReflection);
    this.cache.lastReflection = Date.now();
    await this.saveCache();
  }
  
  // 增量分析：只处理新会话 + 摘要历史
  async incrementalReflect() {
    const newSessions = await this.getUnprocessedSessions();
    const historicalSummary = this.cache.historicalSummary;
    
    // 只对新内容调用 LLM
    const newInsights = await llm.complete(`
      新会话：${newSessions.map(s => s.content).join('\n')}
      历史摘要：${historicalSummary}
      请只分析新增内容和变化趋势。
    `);
    
    return newInsights;
  }
}
```

#### 优化 6：批量 LLM 调用
```javascript
// 使用批处理减少 API 调用次数
async function batchReflect(sessions, batchSize = 5) {
  const batches = [];
  for (let i = 0; i < sessions.length; i += batchSize) {
    batches.push(sessions.slice(i, i + batchSize));
  }
  
  const results = await Promise.all(
    batches.map(batch => 
      llm.complete(`分析以下${batch.length}个会话：${batch.join('\n---\n')}`)
    )
  );
  
  // 最终汇总
  return llm.complete(`汇总以下分析结果：${results.join('\n')}`);
}
```
