// 综合优化后的 session-start.ts 核心逻辑
const fs = require('fs').promises;
const path = require('path');
const os = require('os');

class OptimizedSessionStart {
  constructor() {
    this.skillCache = new Map();
    this.contextCache = null;
    this.cacheTTL = 300000; // 5 分钟
  }

  async sessionStart(context) {
    const cached = this.getCachedContext();
    if (cached) {
      context.additionalContext = cached;
      return context;
    }

    // 并行读取所有技能
    const [usingSuperpowers, soulContent, agentsContent] = await Promise.all([
      this.readSkill('using-superpowers'),
      this.readSkill('soul'),
      this.readSkill('agents')
    ]);

    let additionalContext = "<EXTREMELY_IMPORTANT>\n";
    if (usingSuperpowers) additionalContext += "## Superpowers\n" + usingSuperpowers + "\n\n";
    if (soulContent) additionalContext += "## Identity\n" + soulContent + "\n\n";
    if (agentsContent) additionalContext += "## Goals\n" + agentsContent + "\n\n";
    additionalContext += "</EXTREMELY_IMPORTANT>";

    this.contextCache = { context: additionalContext, timestamp: Date.now() };
    context.additionalContext = additionalContext;
    return context;
  }

  async readSkill(skillName) {
    const cached = this.skillCache.get(skillName);
    if (cached && Date.now() - cached.timestamp < this.cacheTTL) {
      return cached.content;
    }

    const skillPath = this.findSkillPath(skillName);
    if (!skillPath) return null;

    try {
      const content = await fs.readFile(skillPath, "utf8");
      this.skillCache.set(skillName, { content, timestamp: Date.now() });
      return content;
    } catch {
      return null;
    }
  }

  findSkillPath(skillName) {
    const basePaths = [
      path.join(os.homedir(), ".qwen", "skills"),
      path.join(os.homedir(), ".stigmergy", "skills"),
    ];

    for (const basePath of basePaths) {
      for (const fileName of ['SKILL.md', 'skill.md']) {
        const skillPath = path.join(basePath, skillName, fileName);
        if (fs.existsSync(skillPath)) return skillPath;
      }
    }
    return null;
  }

  getCachedContext() {
    if (!this.contextCache) return null;
    if (Date.now() - this.contextCache.timestamp > this.cacheTTL) {
      this.contextCache = null;
      return null;
    }
    return this.contextCache.context;
  }
}

module.exports = new OptimizedSessionStart();
