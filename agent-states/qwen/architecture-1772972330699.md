// 创建统一的路径管理器
class SkillPathManager {
  constructor() {
    this.basePaths = [
      path.join(os.homedir(), '.stigmergy', 'skills'),
      path.join(os.homedir(), '.qwen', 'skills'),
      path.join(process.cwd(), '.agent', 'skills'),
      path.join(process.cwd(), 'skills')
    ];
  }

  async findSkill(skillName) {
    for (const basePath of this.basePaths) {
      const skillPath = path.join(basePath, skillName, 'SKILL.md');
      if (fs.existsSync(skillPath)) {
        return { path: skillPath, source: basePath };
      }
    }
    return null;
  }

  async getAllSkills() {
    const skills = new Map();
    for (const basePath of this.basePaths) {
      if (!fs.existsSync(basePath)) continue;
      const skillDirs = fs.readdirSync(basePath)
        .filter(d => fs.statSync(path.join(basePath, d)).isDirectory());
      for (const dir of skillDirs) {
        if (!skills.has(dir)) {
          skills.set(dir, path.join(basePath, dir));
        }
      }
    }
    return skills;
  }
}

// soul-reflection.js - 实际实现
const fs = require('fs');
const path = require('path');
const os = require('os');

class SoulReflection {
  constructor() {
    this.reflectionDir = path.join(os.homedir(), '.stigmergy', 'reflection');
  }

  async execute(prompt) {
    const cliName = process.env.STIGMERGY_CLI_NAME || 'qwen';
    const historyDir = path.join(os.homedir(), '.stigmergy', 'history', cliName);
    
    // 1. 收集近期工作记录
    const recentSessions = await this.collectRecentSessions(historyDir);
    
    // 2. 构建反思提示
    const reflectionPrompt = this.buildReflectionPrompt(recentSessions);
    
    // 3. 调用 LLM 进行反思 (通过 stigmergy)
    const { execSync } = require('child_process');
    const result = execSync(`stigmergy call "${reflectionPrompt}"`, {
      encoding: 'utf-8',
      maxBuffer: 10 * 1024 * 1024
    });
    
    // 4. 保存反思结果
    const reflectionData = this.parseReflectionResult(result);
    await this.saveReflection(reflectionData);
    
    return this.formatReflectionSummary(reflectionData);
  }

  async collectRecentSessions(historyDir, limit = 10) {
    if (!fs.existsSync(historyDir)) return [];
    
    const files = fs.readdirSync(historyDir)
      .sort()
      .reverse()
      .slice(0, limit);
    
    return files.map(f => ({
      file: f,
      content: fs.readFileSync(path.join(historyDir, f), 'utf-8').substring(0, 5000)
    }));
  }

  buildReflectionPrompt(sessions) {
    return `
你是一个具有深度自我反思能力的 AI 助手。请分析你最近的工作表现：

${sessions.map(s => `--- ${s.file} ---\n${s.content}`).join('\n\n')}

请从以下角度分析：
1. 决策质量
2. 问题解决模式
3. 知识应用
4. 协作方式
5. 自我改进

返回 JSON 格式的反思报告。
`;
  }

  async saveReflection(data) {
    const cliName = process.env.STIGMERGY_CLI_NAME || 'qwen';
    const cliDir = path.join(this.reflectionDir, cliName);
    fs.mkdirSync(cliDir, { recursive: true });
    
    const timestamp = Date.now();
    const filePath = path.join(cliDir, `reflection-${timestamp}.json`);
    fs.writeFileSync(filePath, JSON.stringify(data, null, 2));
    
    // 更新索引
    this.updateIndex(cliDir, { date: data.reflectionDate, file: filePath, score: data.overallScore });
  }
}

module.exports = { execute: async (prompt) => new SoulReflection().execute(prompt) };

// config-manager.js
class ConfigManager {
  constructor() {
    this.configPaths = [
      path.join(os.homedir(), '.stigmergy', 'config.json'),
      path.join(os.homedir(), '.qwen', 'config.json'),
      path.join(process.cwd(), '.stigmergy', 'config.json')
    ];
  }

  async load() {
    const configs = [];
    for (const configPath of this.configPaths) {
      if (fs.existsSync(configPath)) {
        configs.push(JSON.parse(fs.readFileSync(configPath, 'utf-8')));
      }
    }
    return this.mergeConfigs(configs);
  }

  mergeConfigs(configs) {
    // 深度合并配置，优先级：项目 > 用户 > 全局
    return configs.reduce((acc, config) => ({ ...acc, ...config }), {});
  }

  async save(config) {
    const primaryPath = this.configPaths[0];
    fs.mkdirSync(path.dirname(primaryPath), { recursive: true });
    fs.writeFileSync(primaryPath, JSON.stringify(config, null, 2));
  }
}

// skill-dependency-resolver.js
class SkillDependencyResolver {
  async resolveDependencies(skillName) {
    const skill = await this.loadSkill(skillName);
    const dependencies = skill.metadata.dependencies || [];
    
    const resolved = new Set();
    const queue = [...dependencies];
    
    while (queue.length > 0) {
      const dep = queue.shift();
      if (!resolved.has(dep)) {
        resolved.add(dep);
        const depSkill = await this.loadSkill(dep);
        if (depSkill.metadata.dependencies) {
          queue.push(...depSkill.metadata.dependencies);
        }
      }
    }
    
    return Array.from(resolved);
  }
}
