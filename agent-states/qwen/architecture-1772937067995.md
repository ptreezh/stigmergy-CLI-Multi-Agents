class DynamicSecurityPolicy {
  constructor() {
    this.policyStore = new PolicyStore();
    this.riskAnalyzer = new RiskAnalyzer();
    this.auditLogger = new AuditLogger();
  }

  async evaluateCommand(command, context) {
    // 1. 风险评估
    const risk = await this.riskAnalyzer.assess(command, {
      user: context.user,
      session: context.session,
      history: context.commandHistory
    });

    // 2. 策略匹配
    const policy = this.policyStore.match(command);

    // 3. 决策 (允许/拒绝/需要确认)
    if (risk.score > policy.threshold) {
      await this.auditLogger.log('BLOCKED', { command, risk });
      return { allowed: false, reason: risk.reason };
    }

    return { allowed: true };
  }
}

### 🟡 问题 3: Two-Agent Loop 缺少容错机制

**现状**:
```markdown
<!-- two-agent-loop/SKILL.md -->
1. 派发 Subagent → 2. Code Review → 3. 迭代修复 → 4. 循环直到完成
```

**问题**:
- 无最大重试次数限制
- 无超时控制
- 无降级策略
- Subagent 失败后无回滚

**改进建议**:
```javascript
class TwoAgentLoop {
  constructor(options = {}) {
    this.maxRetries = options.maxRetries || 3;
    this.timeout = options.timeout || 120000;
    this.fallbackStrategy = options.fallback || 'abort';
  }

  async execute(task) {
    const circuitBreaker = new CircuitBreaker({
      threshold: 3,
      timeout: 60000
    });

    for (let attempt = 1; attempt <= this.maxRetries; attempt++) {
      try {
        const result = await circuitBreaker.execute(() =>
          this.dispatchSubagent(task, { timeout: this.timeout })
        );

        const review = await this.codeReview(result);
        if (review.passed) return result;

        task = this.enhanceTask(task, review.feedback);
      } catch (error) {
        if (attempt === this.maxRetries) {
          return this.handleFailure(error, task);
        }
      }
    }
  }
}
```

class CrossCliProtocol {
  constructor() {
    this.messageFormat = {
      id: uuid(),
      timestamp: ISO8601,
      source: 'qwen',
      target: 'claude',
      type: 'request' | 'response' | 'error',
      payload: {
        action: 'execute' | 'query' | 'collaborate',
        context: {},
        data: {},
        timeout: 30000
      },
      signature: hmac(payload, secretKey)
    };
  }

  async send(message) {
    const channel = this.getChannel(message.target);
    
    // 1. 序列化
    const serialized = this.serialize(message);
    
    // 2. 发送 (带重试)
    const result = await this.retryableSend(channel, serialized, {
      maxRetries: 3,
      backoff: 'exponential'
    });
    
    // 3. 验证响应
    return this.validateResponse(result, message);
  }
}

// 集中式状态管理
class StigmergyState {
  constructor() {
    this.store = new LevelDB('~/.stigmergy/state.db');
    this.pubsub = new PubSub();
  }

  // 原子操作
  async transaction(operations) {
    const batch = this.store.batch();
    for (const op of operations) {
      batch[op.type](op.key, op.value);
    }
    await batch.write();
    this.pubsub.publish('state:changed', { operations });
  }

  // 状态快照
  async snapshot(label) {
    const state = await this.store.toArray();
    await this.store.put(`snapshots:${label}:${Date.now()}`, state);
  }

  // 时间旅行调试
  async replay(fromSnapshot, toTimestamp) {
    const base = await this.store.get(`snapshots:${fromSnapshot}`);
    const events = await this.getEvents(fromSnapshot, toTimestamp);
    return this.replayEvents(base, events);
  }
}

// 技能版本管理
class SkillVersionManager {
  constructor() {
    this.git = new Git('~/.stigmergy/skills');
  }

  async install(skillName, version = 'latest') {
    // 从技能注册表拉取
    const skill = await this.registry.fetch(skillName, version);
    
    // 语义化版本检查
    const currentVersion = await this.getCurrentVersion(skillName);
    if (!this.isCompatible(currentVersion, skill.version)) {
      throw new BreakingChangeError(skillName, currentVersion, skill.version);
    }
    
    // 原子安装
    await this.git.checkout('main');
    await this.git.add(skill.path);
    await this.git.commit(`Install ${skillName}@${skill.version}`);
    await this.git.tag(`${skillName}@${skill.version}`);
  }

  async rollback(skillName, toVersion) {
    await this.git.checkout(`${skillName}@${toVersion}`);
  }
}
