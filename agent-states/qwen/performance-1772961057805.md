// 优化方案：构建倒排索引
class SkillsIndex {
  constructor() {
    this.keywordMap = new Map();  // keyword -> Set<skillName>
  }
  
  buildIndex(skills) {
    for (const [name, skill] of skills) {
      const keywords = this.extractKeywords(skill.description);
      for (const kw of keywords) {
        if (!this.keywordMap.has(kw)) {
          this.keywordMap.set(kw, new Set());
        }
        this.keywordMap.get(kw).add(name);
      }
    }
  }
  
  search(query) {
    const results = new Set();
    const queryKws = this.extractKeywords(query);
    for (const kw of queryKws) {
      const matches = this.keywordMap.get(kw);
      if (matches) {
        matches.forEach(name => results.add(name));
      }
    }
    return Array.from(results);
  }
}
